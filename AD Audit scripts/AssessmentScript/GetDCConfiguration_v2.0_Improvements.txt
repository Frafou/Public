PowerShell Pipeline Integration:
-------------------------------

# Advanced pipeline processing example
$AssessmentResults = .\GetDCConfiguration_v2.0.ps1 -DomainDN "DC=company,DC=com" -OutputFormat Object
$SecurityIssues = $AssessmentResults.DNSConfiguration | Where-Object { $_.SecurityStatus -eq "Review Required" }
$ComplianceViolations = $AssessmentResults.DNSConfiguration | Where-Object { $_.ComplianceStatus -eq "Non-Compliant" }

if ($SecurityIssues -or $ComplianceViolations) {
    Send-MailMessage -To "security@company.com" -Subject "DC Security Alert" -Body "Domain Controllers require security review"
}

# Automation workflow example
$Domains = @("DC=prod,DC=com", "DC=dev,DC=com", "DC=test,DC=com")
$AllResults = @()

foreach ($Domain in $Domains) {
    $Results = .\GetDCConfiguration_v2.0.ps1 -DomainDN $Domain -Silent -OutputFormat Object
    $AllResults += $Results
}

$AllResults | Export-Csv -Path "C:\Reports\Enterprise_DC_Assessment.csv" -NoTypeInformation
